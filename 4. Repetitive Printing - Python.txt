# Function to print given string 'x' times
string="Geeks"
x=5
def print_fun(string, x):
    print(string * x)

#{

# Driver Code
def main():
    testcases = int(input())
    
    # Loop for testcases
    while(testcases > 0):
        string = input()
        x = int(input())
        print_fun(string, x)
        testcases -= 1

if __name__ == '__main__':
    main()
