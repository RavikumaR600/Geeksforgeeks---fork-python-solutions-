
def checkOddEven(x):
    if(x % 2 == 0):
      # Complete the statement below
      return x

def main():
    
    # Testcase input
    testcases = int(input())
    
    # Looping through testcases
    while(testcases > 0):
        x = int(input())
        
        if(checkOddEven(x)):

        testcases -= 1
